# ===========================================================================
#  5.1 R1 (AR2220)
#  接口:
#    GE0/0/0: 10.1.1.25/30 → InterNet1
#    GE0/0/1: 10.1.1.13/30 → C1 (OSPF Area 0)
#    GE0/0/2: 10.1.1.17/30 → C2 (OSPF Area 0)
# ===========================================================================

<Huawei> system-view
sysname R1

# ---- 接口配置 ----
interface GigabitEthernet0/0/0
 ip address 10.1.1.25 255.255.255.252

interface GigabitEthernet0/0/1
 ip address 10.1.1.13 255.255.255.252

interface GigabitEthernet0/0/2
 ip address 10.1.1.17 255.255.255.252

# ---- ACL + Route-Policy (控制重分布范围) ----
acl 2000
 rule 5 permit source 10.1.1.24 0.0.0.3
 quit
route-policy Internet permit node 10
 if-match acl 2000
 quit

# ---- OSPF Area 0 ----
# Internet 侧 (10.1.1.24/30) 通过 import-route direct 注入为 AS External
ospf 1 router-id 3.3.3.3
 area 0.0.0.0
  network 10.1.1.12 0.0.0.3               # R1↔C1 链路
  network 10.1.1.16 0.0.0.3               # R1↔C2 链路
 import-route direct route-policy Internet  # 仅注入 10.1.1.24/30

save
y


# ===========================================================================
#  5.2 C2 (CE12800)
#  接口:
#    GE1/0/1 (VLAN 201) → R1:        10.1.1.18/30  (OSPF Area 0)
#    GE1/0/2, GE1/0/3 → C1:         Eth-Trunk1 L3, 10.1.1.2/30  (OSPF Area 0)
#    GE1/0/4 (VLAN 102) → D2:        10.1.1.9/30   (静态路由)
#    GE1/0/5 (VLAN 10)  → Cloud1:   192.168.179.1/24 (OSPF, 桥接到VM)
# ===========================================================================

system-view immediately
sysname C2

vlan batch 10 102 201

interface Eth-Trunk1
 undo portswitch
 ip address 10.1.1.2 255.255.255.252
 quit


interface GE1/0/1
 undo shutdown
 port link-type access
 port default vlan 201
 quit

interface GE1/0/2
 undo shutdown
 eth-trunk 1
 quit

interface GE1/0/3
 undo shutdown
 eth-trunk 1
 quit

interface GE1/0/4
 undo shutdown
 port link-type access
 port default vlan 102
 quit

interface GE1/0/5
 undo shutdown
 port link-type access
 port default vlan 10
 quit

# ---- VLANIF ----

interface Vlanif201
 ip address 10.1.1.18 255.255.255.252
 quit

interface Vlanif102
 ip address 10.1.1.9 255.255.255.252
 quit

interface Vlanif10
 ip address 192.168.179.1 255.255.255.0
 quit

# ---- 静态路由 ----
ip route-static 192.168.4.0 255.255.255.0 10.1.1.10
ip route-static 192.168.5.0 255.255.255.0 10.1.1.10
ip route-static 10.1.1.4 255.255.255.252 10.1.1.1   # C1↔D1 直连链路，C1 未通过 OSPF 通告

# ---- ACL + Route-Policy (控制 import-route direct 范围) ----
acl 2000
 rule 5 permit source 192.168.179.0 0.0.0.255
 quit
route-policy CloudNet permit node 10
 if-match acl 2000
 quit

# ---- OSPF Area 0 ----
# Area 0 包含 C2↔R1 (10.1.1.16/30) 和 C2↔C1 Eth-Trunk (10.1.1.0/30)
# Cloud 网段 + D2 下联网段通过 import-route 注入为 AS External
ospf 1 router-id 2.2.2.2
 silent-interface Vlanif102
 area 0.0.0.0
  network 10.1.1.16 0.0.0.3
  network 10.1.1.0 0.0.0.3    # Eth-Trunk1 → C1 (提供冗余路径)
 import-route static
 import-route direct route-policy CloudNet

save
y

## 七、验证命令
###############################################################################

# --- 7.1 直连检查 ---
# R1:
ping 10.1.1.26          # InterNet1
ping 10.1.1.14          # C1
ping 10.1.1.18          # C2

# C1:
ping 10.1.1.13          # R1
ping 10.1.1.2           # C2 Eth-Trunk
ping 10.1.1.6           # D1

# C2:
ping 10.1.1.17          # R1
ping 10.1.1.1           # C1 Eth-Trunk
ping 10.1.1.10          # D2
ping 192.168.179.136    # VM (via Cloud1)

# D1:
ping 10.1.1.5           # C1
ping 192.168.2.2        # PC-A
ping 192.168.3.2        # PC-B

# D2:
ping 10.1.1.9           # C2
ping 192.168.4.2        # PC-C
ping 192.168.5.2        # PC-D

# --- 7.2 Eth-Trunk 检查 ---
# C1 / C2:
display eth-trunk 1
# 预期: 两个成员口 Up

# --- 7.3 OSPF 邻居检查 ----
# R1:
display ospf peer brief
# 预期: 与 C1 (1.1.1.1) 和 C2 (2.2.2.2) 均为 FULL

# C1:
display ospf peer brief
# 预期: 与 R1 (3.3.3.3) 和 C2 (2.2.2.2) 均为 FULL

# C2:
display ospf peer brief
# 预期: 与 R1 (3.3.3.3) 和 C1 (1.1.1.1) 均为 FULL

# --- 7.4 OSPF 路由表检查 ---
# R1:
display ospf routing
# Area 0 内: 10.1.1.12/30, 10.1.1.16/30, 10.1.1.0/30 (Eth-Trunk)
# AS External: 192.168.2/3.0/24 (C1 import), 192.168.4/5.0/24 (C2 import)
#              192.168.179.0/24 (C2 import)

# C1:
display ospf routing
# Area 0 内: 10.1.1.12/30, 10.1.1.16/30, 10.1.1.0/30
# AS External: 10.1.1.24/30 (R1 import), 192.168.4/5.0/24 (C2 import)
#              192.168.179.0/24 (C2 import)

# C2:
display ospf routing
# Area 0 内: 10.1.1.12/30, 10.1.1.16/30, 10.1.1.0/30
# AS External: 10.1.1.24/30 (R1 import), 192.168.2/3.0/24 (C1 import)

# --- 7.5 全局路由表检查 ---
# R1:
display ip routing-table
# 应有到所有 192.168.x.0/24 的 OSPF 路由

# C1 / C2:
display ip routing-table
# 应有到所有网段的 OSPF + 静态 + 直连路由

# --- 7.6 全网端到端 ping 测试 ---
# InterNet1 (10.1.1.26):
ping 192.168.179.136    # VM (via Cloud1)
ping 192.168.2.2        # PC-A
ping 192.168.3.2        # PC-B
ping 192.168.4.2        # PC-C
ping 192.168.5.2        # PC-D

# PC-A (192.168.2.2):
ping 10.1.1.26          # InterNet1
ping 192.168.4.2        # PC-C (跨汇聚)

# PC-B (192.168.3.2):
ping 10.1.1.26          # InterNet1
ping 192.168.5.2        # PC-D (跨汇聚)

# VM (192.168.179.136, 经 Cloud1 桥接):
ping 10.1.1.26          # InterNet1
ping 192.168.2.2        # PC-A

# --- 7.7 链路聚合冗余测试 ---
# C1 上 shutdown 一个 Eth-Trunk 成员口:
interface GE1/0/2
 shutdown
# C1 上检查 OSPF 邻居与 C2 仍保持 FULL
display ospf peer brief
# 恢复:
interface GE1/0/2
 undo shutdown

# --- 7.8 OSPF 冗余测试 (老师验收) ---
# C2 上断开连接 R1 的接口:
interface GE1/0/1
 shutdown
# 从 C2 ping InterNet1 (10.1.1.26)，应仍通 (走 C2→C1 Eth-Trunk→R1→InterNet1)
ping 10.1.1.26
# 恢复:
interface GE1/0/1
 undo shutdown

# --- 7.9 核心链路冗余测试 ---
# R1 同时有两条路径到 C1/C2, shutdown 一条:
interface GigabitEthernet0/0/1
 shutdown
# 从 R1 ping 192.168.4.2 (应仍通, 走 R1→C2→D2)
ping 192.168.4.2
# 从 R1 ping 192.168.2.2 (应仍通, 走 R1→C2→C1→D1, OSPF 自动收敛)
ping 192.168.2.2
# 恢复:
interface GigabitEthernet0/0/1
 undo shutdown
